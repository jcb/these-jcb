%oplist NatList concNat(Nat*) {
  is_fsym(s)         { (s instanceof JconcJNat) }
  get_head(l)        { ((JconcJNat)l).head }
  get_tail(l)        { ((JconcJNat)l).tail }
  is_empty(l)        { ((JconcJNat)l).isEmpty() }
  make_empty()       { new JconcJNat() }
  make_insert(t,l)   { new JconcJNat(t,l) }
}
